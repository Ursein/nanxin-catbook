Step 1 ‚Äî 
  targetCat = SELECT * FROM cat WHERE id = targetCatId
  allCats = SELECT * FROM cat WHERE id != targetCatId AND status = 'ACTIVE'
  
   targetCat 
    colourTagsSet = targetCat.colour_tags.split(';')     //

    location = targetCat.location_area                      // 
    gender = targetCat.gender                               // 

Step 2 ‚Äî 
  globalColours =  colour_tags 
  globalPersonalities =  personality_tags 
  globalLocations =  location_area 
  totalDimensions = len(globalColours) + len(globalPersonalities) 
                    + len(globalLocations) + 3  // 3 = 

Step 3 ‚Äî One-Hot 
  FUNCTION encode(colourTags, personalityTags, location, gender):
    vector = [0] * totalDimensions
    offset = 0
    // 
    FOR tag IN globalColours:
        IF tag IN colourTags: vector[offset] = 1
        offset++
    // 
    FOR tag IN globalPersonalities:
        IF tag IN personalityTags: vector[offset] = 1
        offset++
    // 
    FOR loc IN globalLocations:
        IF loc == location: vector[offset] = 1
        offset++
    //  (MALE/FEMALE/UNKNOWN -> 3)
    genderMap = {'MALE': [1,0,0], 'FEMALE': [0,1,0], 'UNKNOWN': [0,0,1]}
    vector[offset:offset+3] = genderMap[gender]
    RETURN vector

  targetVector = encode(targetCat)
  catVectors = {}
  FOR cat IN allCats:
      catVectors[cat.id] = encode(cat)

Step 4 ‚Äî 
  FUNCTION cosineSimilarity(vecA, vecB):
    dotProduct = SUM(vecA[i] * vecB[i] FOR i IN range(len(vecA)))
    normA = SQRT(SUM(vecA[i]^2 FOR i IN range(len(vecA))))
    normB = SQRT(SUM(vecB[i]^2 FOR i IN range(len(vecB))))
    IF normA == 0 OR normB == 0: RETURN 0
    RETURN dotProduct / (normA * normB)

  similarityScores = {}
  FOR catId, vector IN catVectors:
      similarityScores[catId] = cosineSimilarity(targetVector, vector)

Step 5 ‚Äî 
  // 
  heatData = QUERY:
      SELECT 
          cat_id,
          like_count,       //  photo_like 
          follow_count,     //  cat_follow 
          avg_rating        //  cat_rating 
      FROM cat_rating_stats  // 

  // Min-Max 
  FUNCTION normalize(values):
    minVal = MIN(values)
    maxVal = MAX(values)
    IF maxVal == minVal: RETURN [0.5] * len(values)
    RETURN [(v - minVal) / (maxVal - minVal) FOR v IN values]

  likeNorm = normalize([heatData[catId].like_count FOR catId IN allCats])
  followNorm = normalize([heatData[catId].follow_count FOR catId IN allCats])
  ratingNorm = normalize([heatData[catId].avg_rating FOR catId IN allCats])

  finalScores = {}

      finalScores[catId] = Œ± * similarityScores[catId] 
                         + Œ≤1 * likeNorm[catId]
                         + Œ≤2 * followNorm[catId] 
                         + Œ≤3 * ratingNorm[catId]

Step 6 ‚Äî 
  sortedCats = SORT allCats BY finalScores[catId] DESC

@Service
public class CatSimilarityAlgorithm {
    
    @Autowired private CatRepository catRepository;
    @Autowired private PhotoLikeRepository photoLikeRepository;
    @Autowired private CatFollowRepository catFollowRepository;
    @Autowired private CatRatingRepository catRatingRepository;
    
    public List<RecommendResult> recommend(Long targetCatId, int topN) {
        // 1. 
        Cat target = catRepository.findById(targetCatId);
        List<Cat> allCats = catRepository.findByStatus("ACTIVE");
        
        // 2. 
        FeatureDictionary dict = buildDictionary(allCats);
        
        // 3. 
        double[] targetVector = encode(target, dict);
        
        // 4. 
        Map<Long, Double> simScores = new HashMap<>();
        for (Cat cat : allCats) {
            if (cat.getId().equals(targetCatId)) continue;
            double[] vector = encode(cat, dict);
            simScores.put(cat.getId(), cosineSimilarity(targetVector, vector));
        }
        
        // 5. 
        Map<Long, Double> finalScores = fuseHeatFactors(simScores, allCats);
        
        // 6.  Top-N
        return finalScores.entrySet().stream()
            .sorted(Map.Entry.<Long, Double>comparingByValue().reversed())
            .limit(topN)
            .map(e -> new RecommendResult(e.getKey(), e.getValue()))
            .collect(Collectors.toList());
    }
}
